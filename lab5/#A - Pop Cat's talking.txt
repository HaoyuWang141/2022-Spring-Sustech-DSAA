#include <iostream>
#include <cstring>
using namespace std;

#define MAXSIZE 50000
struct Stack
{
    char data[MAXSIZE]{0};
    int top;

    Stack()
    {
        top = -1;
    }
};

int main()
{
    int n;
    scanf("%d", &n);
    char *str = new char[MAXSIZE * 2];
    for (int i = 0; i < n; i++)
    {
        scanf("%s", str);
        int len = strlen(str);

        Stack stack;

        for (int j = 0; j < len; j++)
        {
            if (str[j] == '(')
            {
                if (stack.top == MAXSIZE)
                {
                    cout << "N0" << endl;
                    goto next;
                }
                else
                {
                    stack.data[++stack.top] = '(';
                }
            }
            else if (str[j] == ')')
            {
                if (stack.top == -1)
                {
                    cout << "N0" << endl;
                    goto next;
                }
                else
                {
                    stack.data[stack.top--] = '\0';
                }
            }
            else
            {
                cerr << "string is not valid" << endl;
            }
        }
        if (stack.top == -1)
            cout << "YE5" << endl;
        else
            cout << "N0" << endl;
    next:;
    }

    return 0;
}
